646  return sqrt(pow(x, 2)+pow(y, 2));

657 inline double vlbi_phase_delegate(double x, double y) {
658  double mag = sqrt(pow(x, 2)+pow(y, 2));
659  double rad = 0.0;
660  if(mag > 0.0) {
661  rad = acos (y / (mag > 0.0 ? mag : 1.0));
662  if(x < 0 && rad != 0)
663  rad = PI*2-rad;
664  }
665  return rad;
666 }
