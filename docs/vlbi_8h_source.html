643  return sqrt(pow(x, 2)+pow(y, 2));

654 inline double vlbi_phase_delegate(double x, double y) {
655  double mag = sqrt(pow(x, 2)+pow(y, 2));
656  double rad = 0.0;
657  if(mag > 0.0) {
658  rad = acos (y / (mag > 0.0 ? mag : 1.0));
659  if(x < 0 && rad != 0)
660  rad = PI*2-rad;
661  }
662  return rad;
663 }
