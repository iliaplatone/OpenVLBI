636  return x*y;
637 }

647 inline double vlbi_magnitude_delegate(double x, double y) {
648  return sqrt(pow(x, 2)+pow(y, 2));
649 }
650 
659 inline double vlbi_phase_delegate(double x, double y) {
660  double mag = sqrt(pow(x, 2)+pow(y, 2));
661  double rad = 0.0;
662  if(mag > 0.0) {
663  rad = acos (y / (mag > 0.0 ? mag : 1.0));
664  if(x < 0 && rad != 0)
665  rad = PI*2-rad;
666  }
667  return rad;
668 }
