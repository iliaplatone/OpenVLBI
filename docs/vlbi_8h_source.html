655 inline double vlbi_phase_delegate(double x, double y) {
656  double mag = sqrt(pow(x, 2)+pow(y, 2));
657  double rad = 0.0;
658  if(mag > 0.0) {
659  rad = acos (y / (mag > 0.0 ? mag : 1.0));
660  if(x < 0 && rad != 0)
661  rad = PI*2-rad;
662  }
663  return rad;
664 }
